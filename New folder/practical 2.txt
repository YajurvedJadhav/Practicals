#include <iostream>
using namespace std;
void selectionSort(float arr[], int n) {
for (int i = 0; i < n - 1; i++) {
int minIndex = i;
for (int j = i + 1; j < n; j++) {
if (arr[j] < arr[minIndex])
minIndex = j;
}
float temp = arr[i];
arr[i] = arr[minIndex];
arr[minIndex] = temp;
}
}
void bubbleSort(float arr[], int n) {
for (int i = 0; i < n - 1; i++) {
for (int j = 0; j < n - i - 1; j++) {
if (arr[j] > arr[j + 1]) {
float temp = arr[j];
arr[j] = arr[j + 1];
arr[j + 1] = temp;
}
}
}
}
void displayTopFive(float arr[], int n) {
cout << "Top Five Scores: ";
int count = 0;
for (int i = n - 1; i >= 0 && count < 5; i--) {
cout << arr[i] << " ";
count++;
}
cout << endl;
}
int main() {
int n;
cout << "Enter number of students: ";
cin >> n;
float arr[100];
cout << "Enter percentages of students: ";
for (int i = 0; i < n; i++) {
cin >> arr[i];
}
float arr1[100], arr2[100];
for (int i = 0; i < n; i++) {
arr1[i] = arr[i];
arr2[i] = arr[i];
}
selectionSort(arr1, n);
cout << "\nUsing Selection Sort:\n";
displayTopFive(arr1, n);
bubbleSort(arr2, n);
cout << "\nUsing Bubble Sort:\n";
displayTopFive(arr2, n);
return 0;
}
